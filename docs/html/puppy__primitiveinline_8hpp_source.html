281 inline void
282 Puppy::Primitive::getArgument(unsigned int inN, void* outResult, Puppy::Context& ioContext)
283 {
284  assert(inN >= 0);
285  assert(inN < mNumberArguments);
286  unsigned int lIndex = ioContext.mCallStack.back() + 1;
287  for(unsigned int i=0; i<inN; ++i) lIndex += (*ioContext.mTree)[lIndex].mSubTreeSize;
288  ioContext.mCallStack.push_back(lIndex);
289  (*ioContext.mTree)[lIndex].mPrimitive->execute(outResult, ioContext);
290  ioContext.mCallStack.pop_back();
291 }
